public class RandomNumberGenerator { public static void main(String[] args) { // Define the range int min = 1; int max = 100; // Create a Random object Random random = new Random(); // Generate a random number within the range int randomNumber = random.nextInt(max - min + 1) + min; // Print the random number System.out.println("Random Number between " + min + " and " + max + ": " + randomNumber); } }
